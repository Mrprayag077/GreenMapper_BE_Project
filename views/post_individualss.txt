    <script>
        function randomNumber(start, end) {
            return Math.floor(Math.random() * (end - start)) + start;
        }

        var likeCount = randomNumber(150000, 300000);

        function createUserComment() {
            var commentBox = document.getElementById("user-comment");
            if (commentBox.value.trim().length == 0) {
                return;
            }
            document.getElementById("custom-comment").innerHTML =
                "<b>You</b> " + commentBox.value;
            commentBox.value = "";
        }

        function updateText() {
            document.getElementById("like-number").innerText = likeCount.toLocaleString();
        }

        function likeImage() {
            likeCount += randomNumber(1, 10);
            updateText();
        }

        updateText();
